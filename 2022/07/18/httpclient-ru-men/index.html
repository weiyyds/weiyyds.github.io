HtttpClient简介

HttpClient 是 Apache Jakarta Common 下的子项目，可以用来提供高效的、最新的、功能丰富的支持 HTTP 协议的客户端编程工具包，并且它支持 HTTP 协议最新的版本和建议。

官方站点：https://hc.apache.org/

官方文档：https://hc.apache.org/httpcomponents-client-4.5.x/current/tutorial/html/

maven包

<dependency>
    <groupId>org.apache.httpcomponents</groupId>
    <artifactId>httpclient</artifactId>
    <version>4.5.2</version>
</dependency>

介绍：

HTTP 协议可能是现在 Internet 上使用得最多、最重要的协议了，越来越多的 Java 应用程序需要直接通过 HTTP 协议来访问网络资源。虽然在 JDK 的 java net包中已经提供了访问 HTTP 协议的基本功能，但是对于大部分应用程序来说，JDK 库本身提供的功能还不够丰富和灵活。HttpClient 是 Apache Jakarta Common 下的子项目，用来提供高效的、最新的、功能丰富的支持 HTTP 协议的客户端编程工具包，并且它支持 HTTP 协议最新的版本和建议。HttpClient 已经应用在很多的项目中，比如 Apache Jakarta 上很著名的另外两个开源项目 Cactus 和 HTMLUnit 都使用了 HttpClient。现在HttpClient最新版本为 HttpClient 4.5 (GA) （2015-09-11）

我们搞爬虫的，主要是用HttpClient模拟浏览器请求第三方站点url，然后响应，获取网页数据，然后用Jsoup来提取我们需要的信息；

HttpClient—Helloword

package com.open111.httpclient;

import org.apache.http.HttpEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

import java.io.IOException;

public class HelloWord {
    public static void main(String[] args) {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("https://www.jb51.net/");
        CloseableHttpResponse response = null;
        try {
            response = httpClient.execute(httpGet);//执行http协议的get请求
        } catch (IOException e) {
            e.printStackTrace();
        }
        HttpEntity entity = response.getEntity();//获取返回实体
        try {
            //获取网页内容并指定编码格式
            String urlResult = EntityUtils.toString(entity, "utf-8");
            System.out.println("网页内容是："+urlResult);
        } catch (IOException e) {
            e.printStackTrace();
        }
        //关闭CloseableHttpResponse
        try {
            response.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
        try {
            //关闭CloseableHttpClient
            httpClient.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

package com.open111.httpclient.chapter02;

import org.apache.http.HttpEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

import java.io.IOException;

/**
 * 模拟浏览器抓取数据
 * 设置响应内容内容content-Type
 */
public class Demo2 {
    public static void main(String[] args) throws IOException {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("http://asset.tuicool.com/assets/pub.js");
        httpGet.setHeader("user-agent","Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/103.0.0.0 Safari/537.36");
        CloseableHttpResponse response = null;
        response = httpClient.execute(httpGet);//执行http协议的get请求
        HttpEntity entity = response.getEntity();//获取返回实体
        //获取网页内容并指定编码格式
        String urlResult = EntityUtils.toString(entity, "utf-8");
        //从entity中获取ContentType
        //获取到了内容类型，作用：过滤掉一些jar包，不需要被jsoup解析
        //application/javascript....还有很多中类型
        System.out.println(entity.getContentType().getValue());
        //System.out.println("网页内容是："+urlResult);
        //关闭CloseableHttpResponse
        response.close();
        //关闭CloseableHttpClient
        httpClient.close();

    }
}

package com.open111.httpclient.chapter03;

import org.apache.commons.io.FileUtils;
import org.apache.http.HttpEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

import java.io.File;
import java.io.IOException;
import java.io.InputStream;

/**
 * 模拟浏览器抓取数据
 * 设置请求头信息user-Agent
 */
public class Demo1 {
    public static void main(String[] args) throws IOException {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("http://mmbiz.qpic.cn/mmbiz_gif/3kMZppQa5zBjthFvJqyMVsnLEq7blXOBQ41lOHEPkHHtp0ntQKGAnSPJvJTUyyOOMqibFk4BRRZSl9bdae8SWQw/640?wx_fmt=gif&wxfrom=5&wx_lazy=1");
        httpGet.setHeader("user-agent","Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/103.0.0.0 Safari/537.36");
        CloseableHttpResponse response = null;
        response = httpClient.execute(httpGet);//执行http协议的get请求
        HttpEntity entity = response.getEntity();//获取返回实体
        if (entity!=null){
            //ContentType:image/gif
            System.out.println("ContentType:"+entity.getContentType().getValue());
            //提取图片
            InputStream inputStream = entity.getContent();
            FileUtils.copyInputStreamToFile(inputStream,new File("D://GitRepository//test_hexo//blog//weiyydsBlog//source//_posts//aaa.gif"));
        }
        //关闭CloseableHttpResponse
        response.close();
        //关闭CloseableHttpClient
        httpClient.close();

    }
}

HttpClient使用代理ip

blog.java1234.com/HttpClient代理ip

在爬取网页的时候，有的目标站点有反爬虫机制，对于频繁访问站点以及规则性访问站点的行为，会采集屏蔽IP措施。

这时候，代理IP就派上用场了。

关于代理IP的话 也分几种 透明代理、匿名代理、混淆代理、高匿代理

透明代理虽然可以直接“隐藏”你的IP地址，但是还是可以从HTTP_X_FORWARDED_FOR来查到你是谁

匿名代理比透明代理进步了一点：别人只能知道你用了代理，无法知道你是谁。

如上，与匿名代理相同，如果使用了混淆代理，别人还是能知道你在用代理，但是会得到一个假的IP地址，伪装的更逼真：-）

可以看出来，高匿代理让别人根本无法发现你是在用代理，所以是最好的选择。

现在爬虫推荐使用高匿代理

好处：让别人根本无法发现你是在用代理，所以是最好的选择

那代理IP 从哪里搞呢 很简单 百度一下，你就知道 一大堆代理IP站点。 一般都会给出一些免费的，但是花点钱搞收费接口更加方便

package com.open111.httpclient.chapter04;

import org.apache.http.HttpEntity;
import org.apache.http.HttpHost;
import org.apache.http.client.config.RequestConfig;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;

import java.io.IOException;

/**
 * 使用代理ip
 */
public class Demo01 {
    public static void main(String[] args) throws IOException {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("http://www.tuicool.com/");
        //设置代理ip
        HttpHost proxy = new HttpHost("112.84.245.57",8080);
        RequestConfig config =RequestConfig.custom().setProxy(proxy).build();
        httpGet.setConfig(config);
        httpGet.setHeader("user-agent","Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/103.0.0.0 Safari/537.36");
        CloseableHttpResponse response = null;
        response = httpClient.execute(httpGet);//执行http协议的get请求
        HttpEntity entity = response.getEntity();//获取返回实体
        //获取网页内容并指定编码格式
        String urlResult = EntityUtils.toString(entity, "utf-8");
        System.out.println("网页内容是："+urlResult);
        //关闭CloseableHttpResponse
        response.close();
        //关闭CloseableHttpClient
        httpClient.close();
    }
}

import java.io.IOException;

/**
 * 使用代理ip,处理连接超时
 */
public class Demo02 {
    public static void main(String[] args) throws IOException {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("https://repo1.maven.org/maven2/com/");
//        HttpGet httpGet = new HttpGet("http://awei.site/");
        //设置代理ip
//        HttpHost proxy = new HttpHost("27.155.220.100",9999);
        HttpHost proxy = new HttpHost("202.111.20.41",80);
        RequestConfig config =RequestConfig.custom()
                .setConnectTimeout(10000)//设置连接超时时间10s=1000ms*10=10000毫秒
                .setSocketTimeout(100)//设置读取数据超时的时间
                .setProxy(proxy).build();

        httpGet.setConfig(config);
        httpGet.setHeader("user-agent","Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/103.0.0.0 Safari/537.36");
        CloseableHttpResponse response = null;
        response = httpClient.execute(httpGet);//执行http协议的get请求
        HttpEntity entity = response.getEntity();//获取返回实体
        //获取网页内容并指定编码格式
        String urlResult = EntityUtils.toString(entity, "utf-8");
        System.out.println("网页内容是："+urlResult);
        //关闭CloseableHttpResponse
        response.close();
        //关闭CloseableHttpClient
        httpClient.close();
    }
}

package com.open111.jsoup;

import org.apache.http.HttpEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;

import java.io.IOException;

/**
 * Jsoup获取DOM元素属性值
 */
public class Demo04 {
    public static void main(String[] args) throws IOException {
        //创建httpClient客户端实例
        CloseableHttpClient httpClient = HttpClients.createDefault();
        //创建httpGet实例
        HttpGet httpGet = new HttpGet("http://www.cnblogs.com/");
        CloseableHttpResponse response = null;
        response = httpClient.execute(httpGet);//执行http协议的get请求
        HttpEntity entity = response.getEntity();//获取返回实体
        //获取网页内容并指定编码格式
        String content = EntityUtils.toString(entity, "utf-8");
        //关闭CloseableHttpResponse
        response.close();
        //关闭CloseableHttpClient
        httpClient.close();
        Document doc = Jsoup.parse(content);//解析网页得到文档对象
        Element element = doc.getElementsByTag("title").first();//获取tag是title的所有dom元素
        String title = element.text();//返回元素的文本
        System.out.println("网页标题是："+title);
        Elements linkElements = doc.select("#post_list .post-item .post-item-body .post-item-text .post-item-title");//通过css选择器查找对应DOM节点
        for (Element linkElement : linkElements) {
            //获取DOM节点属性为href的值，通过attr方法
            System.out.println("博客地址："+linkElement.attr("href"));
        }
        //获取友情链接
        Element link = doc.select("#friend_link").first();
        System.out.println("纯文本："+link.text());
        System.out.println("Html:"+link.html());


    }
}
