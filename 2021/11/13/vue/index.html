        }
        Object.defineProperty(person,'age',{
            // value:18,
            // enumerable:true,//设置属性可以被枚举
            // writable:true,//设置属性可以被修改
            // configurable:true,//设置属性能够被删除 delete person.age

            //当有人读取person里的age时，get函数(getter)就会被调用，且返回值就是age的值
            get(){
                return number
            },
            //当有人修改person里的age时，set函数(setter)就会被调用，且会收到修改的具体值
            set(value){
                console.log('有人修改了age的值，且修改的值是：',value);
                number = value
            }

        })
        console.log(person);
    </script>

        firstName(newValue){

                console.log(this);
                this.fullName= newValue + "-"+this.lastName;

        } ,
        lastName(newValue){

        }

key的原理

<!DOCTYPE html>

    <!-- 
        面试题：react、vue中的key有什么作用（key的内部原理）
        1.虚拟Dom中key的作用
           key是虚拟DOM对象的标识，当数据发生变化时，vue会根据【新数据】生成【新的虚拟DOM】
           随后vue进行【新虚拟DOM】与【旧虚拟DOM】的差异比较，比较规则如下：
        2.对比规则
           （1）新虚拟DOM中找到了与旧虚拟DOM相同的key
                若虚拟DOM中内容不变，直接使用之前的真实DOM
                若虚拟DOM内容改变，则生成新的真实DOM，随后替换页面中之前的真实DOM
           （2）新虚拟DOM中未找到了与旧虚拟DOM相同的key
           创建新的真实DOM，随后渲染到页面
        3.用index作为key可能会引发的问题
            1.若对数据进行逆序添加删除操等破坏顺序操作
              会产生没有必要的真实DOM更新==》界面效果没问题，但效率低
            2.如果结构中还包含有输入类的DOM
               会产生错误DOM更新==》界面有问题
       4.开发中如何选择key?
            1.最好使用每条数据的唯一标识作为key，比如id，手机号、身份证号、学号等唯一值
            2.如果不存在对数据的逆序添加、逆序删除等破坏顺序操作，仅用于渲染列表用于展示，
              使用index作为key是没有问题的
     -->
    <div id="root">
        <!-- 遍历数组 -->
        <h2>人员列表</h2>
        <button v-on:click.once="add">添加一个老六</button>

            <!-- <li v-for="p in persons" :key="p.id">{{p.name}}-{{p.age}}</li> -->
            <li v-for="(p,index) of persons" :key="p.id">
                {{p.name}}-{{p.age}}
                <input type="text" name="" id="">
            </li>
        </ul>
       
    </div>
</body>

            <!-- <li v-for="p in persons" :key="p.id">{{p.name}}-{{p.age}}</li> -->

<script type="text/javascript">
    //vscode的折叠#region 与#endregion
    //#region 
    /*
    //设置为 false 以阻止 vue 在启动时生成生产提示。
    Vue.config.productionTip = false;
    //创建vue实例
    //用watch实现
   new Vue({
       el:"#root",
       data:{
          keyword:'',
          persons:[
              {id:'001',name:'马冬梅',age:19,sex:'女'},
              {id:'002',name:'周冬雨',age:20,sex:'女'},
              {id:'003',name:'周杰伦',age:21,sex:'男'},
              {id:'004',name:'温兆伦',age:22,sex:'男'}
          ],
          filpersons:[]
      
       },
       watch:{
           //方式一完整形式
        keyword:{
            immediate:true,
            handler(val){
                this.filpersons=this.persons.filter((p)=>{
                             return p.name.indexOf(val)!=-1;
                         })
            }
        }

        <input type="text" placeholder="请输入名字" v-model="keyword">
        <button v-on:click="sortType=2">年龄升序</button>
        <button v-on:click="sortType=1">年龄降序</button>
        <button v-on:click="sortType=0">原顺序</button>
        <ul>
            <!-- <li v-for="p in persons" :key="p.id">{{p.name}}-{{p.age}}</li> -->
            <li v-for="(p,index) of filpersons">
                {{p.name}}-{{p.age}}--{{p.sex}}
            </li>

<script type="text/javascript">
 
    //用computed实现
    Vue.config.productionTip = false;
    //创建vue实例
    //用watch实现
   new Vue({
       el:"#root",
       data:{
          keyword:'',
          sortType:0,//0 :原顺序 1:降序 2:升序
          persons:[
              {id:'001',name:'马冬梅',age:30,sex:'女'},
              {id:'002',name:'周冬雨',age:34,sex:'女'},
              {id:'003',name:'周杰伦',age:19,sex:'男'},
              {id:'004',name:'温兆伦',age:15,sex:'男'}
          ]
       },
       computed:{
           filpersons(){
            const arr= this.persons.filter((p)=>{
                             return p.name.indexOf(this.keyword)!=-1;
                         })
            //判断一下是否需要排序
            if(this.sortType){
                arr.sort((p1,p2)=>{
                    return this.sortType===1?p2.age-p1.age:p1.age-p2.age;//a-b代表升序，b-a代表降序
                })
            }
            return arr;
           }
       }
      
     
    
    });
